# Toast 和 ConfirmDialog 使用示例

## Toast 通知

### 基本使用

```typescript
'use client'

import { useToast } from '@/contexts/ToastContext'

export default function MyComponent() {
  const toast = useToast()

  const handleSuccess = () => {
    toast.success('操作成功！')
  }

  const handleError = () => {
    toast.error('操作失败，请重试')
  }

  const handleInfo = () => {
    toast.info('这是一条提示信息')
  }

  const handleWarning = () => {
    toast.warning('警告：此操作不可撤销')
  }

  return (
    <div>
      <button onClick={handleSuccess}>成功提示</button>
      <button onClick={handleError}>错误提示</button>
      <button onClick={handleInfo}>信息提示</button>
      <button onClick={handleWarning}>警告提示</button>
    </div>
  )
}
```

### 替换 alert()

```typescript
// ❌ 旧代码
if (data.success) {
  alert('保存成功！')
}

// ✅ 新代码
if (data.success) {
  toast.success('保存成功！')
}
```

## ConfirmDialog 确认对话框

### 基本使用

```typescript
'use client'

import { useState } from 'react'
import ConfirmDialog from '@/components/common/ConfirmDialog'

export default function MyComponent() {
  const [showDialog, setShowDialog] = useState(false)

  const handleDelete = () => {
    setShowDialog(true)
  }

  const confirmDelete = async () => {
    // 执行删除操作
    await fetch('/api/delete', { method: 'DELETE' })
    setShowDialog(false)
  }

  return (
    <>
      <button onClick={handleDelete}>删除</button>

      <ConfirmDialog
        isOpen={showDialog}
        title="确认删除"
        message="删除后无法恢复，确定要删除吗？"
        confirmText="删除"
        cancelText="取消"
        type="danger"
        onConfirm={confirmDelete}
        onCancel={() => setShowDialog(false)}
      />
    </>
  )
}
```

### 不同类型

```typescript
// 危险操作（红色）
<ConfirmDialog
  type="danger"
  title="确认删除"
  message="此操作不可撤销"
  // ...
/>

// 警告操作（黄色）
<ConfirmDialog
  type="warning"
  title="注意"
  message="请确认此操作"
  // ...
/>

// 信息确认（蓝色）
<ConfirmDialog
  type="info"
  title="提示"
  message="是否继续？"
  // ...
/>
```
